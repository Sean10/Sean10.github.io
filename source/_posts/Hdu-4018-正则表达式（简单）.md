```
#include <cstdio>
#include <cstdlib>
#include <cstring>

int main(){
    //freopen("in.txt","r",stdin);
    char str[100];
    char ch[100];
    int n;

    scanf("%d",&n);
    for(int j = 1;j <= n;j++){
        scanf("%s",str);
        for(int i = 0;str[i+2] != '\0';i++){
            if(str[i] == '/'){
                if(str[i+1] == '/'){
                    strcpy(ch,&str[i+2]);
                    break;
                }

            }
        }
        for(int i = 0;ch[i] != '\0';i++){
            if(ch[i] == '/' || ch[i] == ':'){
                ch[i] = '\0';
                break;
            }
        }
        printf("Case #%d: %s\n",j,ch);
    }
    return 0;
}
```
